#include <stdio.h>

int main() {
    int n, nmaior;
    
    printf("Qual a ordem da matriz?\n");
    scanf("%d", &n);
    
    int matriz[n][n];
    
for(int i = 0; i<n; i++){
    for(int j = 0; j < n; j++){
            printf("Elemento[%d, %d]\n", i, j);
            scanf("%d", &matriz[i][j]);
    }
}
   for (int i = 0; i < n; i++) {
        int nmaior = matriz[i][0];
        for (int j = 1; j < n; j++) {
            if (matriz[i][j] > nmaior) {
                nmaior = matriz[i][j];
            }
        }
        printf("- O maior elemento da linha %d é:  %d\n", i + 1, nmaior);
   }           
        
    
    return 0;
}
